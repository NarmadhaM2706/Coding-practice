class Solution {
    public boolean detectCapitalUse(String word) {
        int count = 0;

        for (char c : word.toCharArray()) {
            if (Character.isUpperCase(c)) {
                count++;
            }
        }

        return count == word.length() ||      // all uppercase
               count == 0 ||                  // all lowercase
               (count == 1 && Character.isUpperCase(word.charAt(0))); // only first uppercase
    }
}
